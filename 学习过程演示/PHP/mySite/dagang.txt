四、SQL语句准备
--创建数据库
create database mySite;
--切换数据库
use mySite;
--用户表
create table user(
	id int unsigned not null auto_increment primary key comment "id",
	username varchar(30) not null comment "用户名",
	password char(32) not null comment "密码"
);
--关键字表
create table keywords(
	id int unsigned not null auto_increment primary key comment "id",
	keyword varchar(100) not null comment "关键字",
	frequency int unsigned not null comment "被搜索次数"
);
--轮播图表
create table carousel(
	id int unsigned not null auto_increment primary key comment "id",
	img varchar(100) not null comment "轮播图图片",
	shop_id int unsigned not null comment "商品id"
);
--广告表
create table advertising(
	id int unsigned not null auto_increment primary key comment "id",
	img varchar(100) not null comment "广告图片",
	place tinyint unsigned not null comment "广告位置"
);
--商品展示表
create table shop(
	id int unsigned not null auto_increment primary key comment "id",
	name varchar(50) not null comment "商品名称",
	intro varchar(100) not null comment "商品简介",
	img varchar(100) not null comment "商品图片",
	currentPrice float unsigned not null comment "现价",
	originalPrice float unsigned not null comment "原价",
	salesVolume int unsigned not null default "1" comment "销量",
	repertory int unsigned not null comment "库存",
	goodsShelf tinyint unsigned not null comment "上下架",
	mail tinyint unsigned not null comment "包邮",
	integral int unsigned not null comment "商品积分",
	award int unsigned not null comment "奖励积分",
	address varchar(50) not null comment "商品地址",
	keywords_id int unsigned not null comment "关键字id",
	LClassify_id int unsigned not null comment "小分类id",
	evaluate_id int unsigned not null comment "评论id",
	priceSection_id int unsigned not null comment "价格区间表id"
);
--导航栏内容表
create table nav(
	id int unsigned not null auto_increment primary key comment "id",
	content varchar(100) not null comment "内容",
	BClassify_id int unsigned not null comment "大分类id"
);
--大分类表
create table BClassify(
	id int unsigned not null auto_increment primary key comment "id",
	name varchar(30) not null comment "内容"
);
--小分类表
create table LClassify(
	id int unsigned not null auto_increment primary key comment "id",
	name varchar(30) not null comment "内容",
	BClassify_id int unsigned not null comment "大分类id"
);
--价格区间表
create table priceSection(
	id int unsigned not null auto_increment primary key comment "id",
	section varchar(30) not null comment "价格区间"
);
--评论表
create table evaluate(
	id int unsigned not null auto_increment primary key comment "id",
	content varchar(300) not null comment "评价内容",
	time varchar(50) not null comment "评价时间",
	grade tinyint unsigned not null comment "等级",
	user_id int unsigned not null comment "用户",
	shop_id int unsigned not null comment "评价的商品"
);
--购物车表
create table shopping(
	id int unsigned not null auto_increment primary key comment "id",
	user_id int unsigned not null comment "用户",
	shop_id int unsigned not null comment "商品"
);
--订单表
create table indent(
	id int unsigned not null auto_increment primary key comment "id",
	orderNumber int unsigned not null comment "订单号",
	states_id int unsigned not null comment "订单状态",
	address_id int unsigned not null comment "收获地址",
	user_id int unsigned not null comment "用户",
	shop_id int unsigned not null comment "商品",
	message_id int unsigned not null comment "买家留言"
);
--订单状态表
create table states(
	id int unsigned not null auto_increment primary key comment "id",
	state varchar(30) not null comment "订单状态"
);
--收藏表
create table collect(
	id int unsigned not null auto_increment primary key comment "id",
	time varchar(30) not null comment "收藏时间",
	user_id int unsigned not null comment "用户",
	shop_id int unsigned not null comment "商品"
);
--收获地址表
create table address(
	id int unsigned not null auto_increment primary key comment "id",
	name varchar(30) not null comment "收件人",
	phone int unsigned not null comment "联系电话",
	address varchar(50) not null comment "收货地址",
	user_id int unsigned not null comment "用户"
);
--基本资料表
create table datum(
	id int unsigned not null auto_increment primary key comment "id",
	img varchar(100) not null comment "头像",
	sex tinyint unsigned not null comment "性别",
	award int unsigned not null comment "积分",
	phone int unsigned not null comment "联系电话",
	QQ int unsigned not null comment "QQ",
	birthday char(10) not null comment "生日",
	address varchar(50) not null comment "所在地区",
	user_id int unsigned not null comment "用户"
);
--积分商品表
create table awardShop(
	id int unsigned not null auto_increment primary key comment "id",
	name varchar(50) not null comment "商品名称",
	intro varchar(100) not null comment "商品简介",
	img varchar(100) not null comment "商品图片",
	price int unsigned not null comment "所需积分",
	salesVolume int unsigned not null default "1" comment "销量",
	repertory int unsigned not null comment "库存",
	goodsShelf tinyint unsigned not null comment "上下架",
	mail tinyint unsigned not null comment "包邮",
	award int unsigned not null comment "奖励积分",
	address varchar(50) not null comment "商品地址",
	evaluate_id int unsigned not null comment "评论id",
);
--买家留言表
create table message(
	id int unsigned not null auto_increment primary key comment "id",
	content varchar(100) not null comment "留言内容",
	user_id int unsigned not null comment "用户",
	shop_id int unsigned not null comment "购买的商品"
);
